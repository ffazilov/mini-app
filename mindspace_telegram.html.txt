function addMsg(role,text){
  const msgs=document.getElementById('messages');
  if(role==='mood'){msgs.insertAdjacentHTML('beforeend',text);msgs.scrollTop=msgs.scrollHeight;return}
  const isUser=role==='user';
  const isHuman=role==='human';
  const avClass=isHuman?'human-sm':'ai-sm';
  const avIcon=isHuman?'🙋':'🤖';
  const bubClass=isUser?'user-b':isHuman?'human-b':'ai-b';
  const html=`<div class="msg-row ${isUser?'user':''}">
    ${!isUser?`<div class="msg-avatar-sm ${avClass}">${avIcon}</div>`:''}
    <div class="bubble ${bubClass}">${text}</div>
  </div>`;
  msgs.insertAdjacentHTML('beforeend',html);
  msgs.scrollTop=msgs.scrollHeight;
}

  breathIv=setInterval(()=>{
    breathCount--;
    document.getElementById('bNum').textContent=breathCount;
    if(breathCount<=0){clearInterval(breathIv);breathIdx++;if(breathIdx%3===0)breathCycles++;runBreath()}
